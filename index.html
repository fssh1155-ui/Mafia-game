// 1. إنشاء الغرفة (النسخة المعدلة لكشف الأخطاء) function createRoom() { myName = document.getElementById('playerName').value.trim(); if (!myName) { showNotification("الرجاء إدخال اسمك أولاً!"); return; } currentRoom = Math.floor(1000 + Math.random() * 9000).toString(); // إضافة إشعار لنعرف أن الزر استجاب فعلاً للضغطة showNotification("جاري الاتصال بالسيرفر..."); // تسجيل بيانات الغرفة والمضيف db.ref('rooms/' + currentRoom).set({ host: myName, state: 'lobby', players: { [myName]: { name: myName, role: "waiting", isAlive: true } } }).then(() => { document.getElementById('displayRoomCode').innerText = currentRoom; showNotification("✅ تم إنشاء الغرفة بنجاح"); showScreen('lobby-screen'); listenToRoom(); }).catch((error) => { // هنا سيظهر لك الخطأ الحقيقي بدلاً من السكوت showNotification("❌ فشل الاتصال: انظر لـ Console"); console.error("تفاصيل الخطأ:", error.message); }); }
